```python
class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:

        l = 1
        r = max(piles)  # 10**9 - 1 also works, but you donâ€™t need to eat faster than the largest pile

        # bisect_left since we want the minimum speed
        while l <= r:
            speed = (l + r) // 2

            duration = 0
            for pile in piles:
                duration += -(-pile // speed)   # math.ceil() equivalent
                if duration > h:    # early break saves a bit of time on longer piles lists
                    break

            if duration <= h:   # if condition satisfies
                r = speed - 1   # we try a smaller one (bisect_left shrinks r; bisect_right bumps l)
            else:
                l = speed + 1   

        return l    # l for bisect_left, r for bisect_right
```
