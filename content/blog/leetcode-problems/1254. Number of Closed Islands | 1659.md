```python
class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        # try not to use union find, duh
        # idea:
        # traverse grid normally, mark and track visited locations
        # when greeted a 0, we start a bfs search and keep track of all the connected 0s
        # island_number += 1
        # continue to traverse grid, skip any previously visited location
        # until the bottom right most square

        island_number = 0
        visited = set()  # (0,0), (0,1), ...
        m, n = len(grid), len(grid[0])  # height, left-to-right

        for i in range(m):
            for j in range(n):
                square = (i, j)
                if square in visited or grid[i][j] == 1:
                    continue

                q = deque([square])
                is_closed_island = True

                while q:
                    tile = q.popleft()
                    visited.add(tile)
                    x, y = tile
                    # if edge land
                    if is_closed_island and (x in [0, m - 1] or y in [0, n - 1]):
                        is_closed_island = False
                    # right
                    if y + 1 < n and (x, y + 1) not in visited and grid[x][y + 1] == 0:
                        q.append((x, y + 1))
                    # left
                    if y - 1 > -1 and (x, y - 1) not in visited and grid[x][y - 1] == 0:
                        q.append((x, y - 1))
                    # down
                    if x + 1 < m and (x + 1, y) not in visited and grid[x + 1][y] == 0:
                        q.append((x + 1, y))
                    # up
                    if x - 1 > -1 and (x - 1, y) not in visited and grid[x - 1][y] == 0:
                        q.append((x - 1, y))

                if is_closed_island:
                    island_number += 1

        return island_number
```

It's O(M*N) no matter which method you use.
