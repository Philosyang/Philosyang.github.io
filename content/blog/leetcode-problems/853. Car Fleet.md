```python
class Solution:
    def carFleet(self, target: int, position: List[int], speed: List[int]) -> int:
        # ideology:
        # if a car takes less time to reach the target
        # compared to the car in front
        # that means they will be in a fleet midway

        pos_time = [(p, (target - p) / s) for p, s in zip(position, speed)]
        pos_time.sort(reverse=True)

        fleet_count = 0
        last_duration = 0.0

        for pos, time in pos_time:
            if time > last_duration:
                fleet_count += 1
                last_duration = time

        return fleet_count
```
