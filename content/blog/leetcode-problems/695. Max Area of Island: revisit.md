# flip 1s to 0s as we dfs


class Solution:
    def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
        if not grid or not grid[0]:
            return 0
        m, n = len(grid), len(grid[0])

        max_area = 0
        current_area = 0
        directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]

        def traversal(x: int, y: int):
            nonlocal current_area
            if grid[x][y] == 0:
                return
            grid[x][y] = 0
            current_area += 1

            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < m and 0 <= ny < n:
                    traversal(nx, ny)
            return

        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    current_area = 0
                    traversal(i, j)
                    max_area = max(max_area, current_area)

        return max_area
