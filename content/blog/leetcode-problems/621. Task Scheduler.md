The idea is not easy to be discovered.

```python
# greedy: start with the task that appears the most
# BABABA or ABABAB
# BABAB
# B_BABAB is optimal


class Solution:
    def leastInterval(self, tasks: List[str], n: int) -> int:
        freq = Counter(tasks)
        heap = [-cnt for cnt in freq.values()]
        heapq.heapify(heap)
        cd = deque()

        timestamp = 0

        while heap or cd:
            timestamp += 1

            while cd and cd[0][0] <= timestamp:
                _, ready_count = cd.popleft()
                heapq.heappush(heap, ready_count)

            if heap:
                count = heapq.heappop(heap)
                count += 1
                if count != 0:
                    cd.append((timestamp + n + 1, count))

        return timestamp
```
