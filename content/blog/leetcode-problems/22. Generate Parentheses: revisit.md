```python
# idea: allow both when
# number of ( > number of )
# else only allow (

class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        result = []

        def bt(food: str, lefts: int, rights: int) -> None:
            if lefts == rights == n:
                result.append(food)
                return

            if lefts < n:
                bt(food + "(", lefts + 1, rights)
            if lefts > rights:
                bt(food + ")", lefts, rights + 1)

            return

        bt("", 0, 0)

        return result
```
