```python
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        items = set(nums)
        best = 0

        for num in items:
            if num - 1 not in items:
                x = num
                while x + 1 in items:
                    x += 1
                best = max(best, x - num + 1)
        return best
```
