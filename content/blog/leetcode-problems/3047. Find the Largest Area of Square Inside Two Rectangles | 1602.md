```python
class Solution:
    def largestSquareArea(
        self, bottomLeft: List[List[int]], topRight: List[List[int]]
    ) -> int:
        # it's okay to be O(N**2)!

        def calculateSquareArea(rec0: List[List[int]], rec1: List[List[int]]) -> int:
            # see leetcode 223.
            # rec0 = [leftbottom, righttop] = [[1,1], [3,3]]
            # rec1 = [leftbottom, righttop] = [[2,2], [4,4]]

            maxleft = max(rec0[0][0], rec1[0][0])
            minright = min(rec0[1][0], rec1[1][0])
            maxbottom = max(rec0[0][1], rec1[0][1])
            mintop = min(rec0[1][1], rec1[1][1])

            if maxleft >= minright or maxbottom >= mintop:  # no overlap
                return 0

            # return square of the shorter overlap
            return min(minright - maxleft, mintop - maxbottom) ** 2

        n = len(bottomLeft)
        max_area = 0
        rectangles = []

        for i in range(n):
            rectangles.append([bottomLeft[i], topRight[i]])

        # print(rectangles)

        for i, rec0 in enumerate(rectangles[:-1]):
            for rec1 in rectangles[i + 1 :]:
                area = calculateSquareArea(rec0, rec1)
                # print(rec0, rec1, area)
                if area > max_area:
                    max_area = area

        return max_area
```
