```python
class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        result = []
        n = len(nums)

        def bt(subset: List[int], start: int) -> None:
            result.append(subset[:])
            for i in range(start, n):
                subset.append(nums[i])
                bt(subset, i + 1)
                subset.pop()

        bt([], 0)

        return result
```
