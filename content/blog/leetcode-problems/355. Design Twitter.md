class Twitter:

    def __init__(self):
        self.timestamp = (
            -1
        )  # question doesn't guarantee monotonic tweetId, we increment negatively for min heap
        self.follows = {}  # follower: set(followees)
        self.posts = {}  # user: list(timestamp, tweetId) auto sorted

    def postTweet(self, userId: int, tweetId: int) -> None:
        if userId in self.posts:
            self.posts[userId].append((self.timestamp, tweetId))
        else:
            self.posts[userId] = [(self.timestamp, tweetId)]
        self.timestamp -= 1

    def getNewsFeed(self, userId: int) -> List[int]:
        newsfeed = []
        persons = [userId]
        persons.extend(
            self.follows.get(userId, [])
        )  # constraints mentioned that 'A user cannot follow himself.'
        heap = []
        for p in persons:
            if p in self.posts:  # check, in case the followee never posted
                # timestamp, tweetId, from_userId, position_in_users_post_list
                heap.append((self.posts[p][-1][0], self.posts[p][-1][1], p, -1))
        heapq.heapify(heap)

        while len(newsfeed) < 10 and heap:
            timestamp, tweetId, from_userId, pos = heapq.heappop(heap)
            newsfeed.append(tweetId)
            if len(self.posts[from_userId]) >= -pos + 1:
                heapq.heappush(
                    heap,
                    (
                        self.posts[from_userId][pos - 1][0],
                        self.posts[from_userId][pos - 1][1],
                        from_userId,
                        pos - 1,
                    ),
                )

        return newsfeed

    def follow(self, followerId: int, followeeId: int) -> None:
        if followerId in self.follows:
            self.follows[followerId].add(followeeId)
        else:
            self.follows[followerId] = set([followeeId])

    def unfollow(self, followerId: int, followeeId: int) -> None:
        if followerId in self.follows and followeeId in self.follows[followerId]:
            self.follows[followerId].remove(followeeId)
