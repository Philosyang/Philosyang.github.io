class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        # 1. search in first col
        col = [row[0] for row in matrix]

        n = len(col)
        l = 0
        r = n - 1

        while l <= r:
            m = (l + r) // 2

            if col[m] == target:
                return True
            elif col[m] > target:
                r = m - 1
            else:
                l = m + 1

        # 2. search in that row
        row = matrix[r]

        n = len(row)
        l = 0
        r = n - 1

        while l <= r:
            m = (l + r) // 2

            if row[m] == target:
                return True
            elif row[m] > target:
                r = m - 1
            else:
                l = m + 1

        return False
