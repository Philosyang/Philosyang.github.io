```python
# "All the timestamps timestamp of set are strictly increasing."
# is vital to enable binary search


class TimeMap:

    def __init__(self):
        self.storage = defaultdict(list)

    def set(self, key: str, value: str, timestamp: int) -> None:
        self.storage[key].append((timestamp, value))

    def get(self, key: str, timestamp: int) -> str:
        n = len(self.storage[key])
        if not n:
            return ""

        l = 0
        r = n - 1

        # [0,0,1,1,1,7,8] we want latest <= 4
        #  l     m     r; m < target therefore l = m
        #        l   m r; m > target therefore r = m - 1
        #          l
        #  l/r is correct index

        # [0,0,1,1,4,4,8] we want latest <= 4
        #  l     m     r; m < target therefore l = m
        #        l   m r; m = target therefore l = m
        #            l r; m > target therefore r = m - 1
        #            l
        #  l/r is correct index

        # [5,5,5] we want latest <= 4
        #  l m r; m > target therefore r = m - 1
        #  l    ; m > target therefore r = m - 1
        #  l == 0, r < l, we should catch this
        while l < r:

            m = (l + r + 1) // 2

            if self.storage[key][m][0] <= timestamp:
                l = m
            else:
                r = m - 1

        # print(l, r, self.storage[key], timestamp)
        return "" if self.storage[key][l][0] > timestamp else self.storage[key][l][1]


# Your TimeMap object will be instantiated and called as such:
# obj = TimeMap()
# obj.set(key,value,timestamp)
# param_2 = obj.get(key,timestamp)
```

Let me think of a way to force the use of a conventional floor'ed `m`.

```python
class TimeMap:

    def __init__(self):
        self.tmap = defaultdict(list)

    def set(self, key: str, value: str, timestamp: int) -> None:
        self.tmap[key].append((timestamp, value))

    def get(self, key: str, timestamp: int) -> str:
        if not self.tmap[key] or self.tmap[key][0][0] > timestamp:
            return ""

        n = len(self.tmap[key])
        l = 0
        r = n - 1

        # [1,1,4,4,7,7] find latest <= 4
        #  l   m     r; m <=, l = m + 1
        #        l m r; m > , r = m - 1
        #        3    ; , <=, l = m + 1
        # r is correct

        # [1,4] find latest <= 1
        #  l r; m <=, l = m + 1
        #    3; m > , r = m - 1
        # r is correct

        while l <= r:
            m = (l + r) // 2

            if self.tmap[key][m][0] <= timestamp:
                l = m + 1
            else:
                r = m - 1

        return self.tmap[key][r][1]


# Your TimeMap object will be instantiated and called as such:
# obj = TimeMap()
# obj.set(key,value,timestamp)
# param_2 = obj.get(key,timestamp)
```
