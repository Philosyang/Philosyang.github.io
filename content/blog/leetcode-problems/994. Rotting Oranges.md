Classic oranges.

```python
# iconic level based bfs


class Solution:
    def orangesRotting(self, grid: List[List[int]]) -> int:
        if not grid or not grid[0]:
            return 0

        m, n = len(grid), len(grid[0])
        rotten = deque()
        fresh_sum = 0  # easy check whether any fresh left
        minutes = 0
        directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]

        for i in range(m):
            for j in range(n):
                cell = grid[i][j]
                if cell == 2:
                    rotten.append((i, j))
                elif cell == 1:
                    fresh_sum += 1

        while rotten:
            for _ in range(len(rotten)):
                x, y = rotten.popleft()
                for dx, dy in directions:
                    nx, ny = x + dx, y + dy
                    # if rot-able
                    if 0 <= nx < m and 0 <= ny < n and grid[nx][ny] == 1:
                        fresh_sum -= 1
                        grid[nx][ny] = 2
                        rotten.append((nx, ny))
            if rotten:
                minutes += 1

        return minutes if not fresh_sum else -1
```
