```python
class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        # 3,5,7,9
        # first of all, sum must be even
        # if you pick the largest, you should then pick the smallest?
        # the sum is settled before picking
        # question turns into: find combination that sums up to X
        # recusion/dfs?
        # for each step, either pick or skip

        n = len(nums)
        goal = sum(nums)
        if goal % 2:  # x is odd
            return False
        else:
            goal //= 2

        @lru_cache(None)
        def dfs(running_sum: int, idx: int):
            # print(running_sum, idx)
            if idx >= n or running_sum > goal:
                return False
            elif running_sum == goal:
                return True
            return dfs(running_sum + nums[idx], idx + 1) or dfs(running_sum, idx + 1)
        
        return dfs(0, 0)
```
