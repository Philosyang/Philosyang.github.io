Feels like a cheeseburger. One thing that could be improved is to use a set() as a counter instead of a dict.

```python
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        # classic sliding window
        # expand r until dupes
        # shrink l until no dupes
        # update longest every step

        n = len(s)
        if not n:
            return 0

        l = 0
        r = 0  # inclusive
        longest = 1
        deduper = {s[0]: 1}

        while r < n - 1:
            r += 1
            if deduper.get(s[r], -1) < 1:
                deduper[s[r]] = 1
                longest = max(longest, r - l + 1)
            else:
                deduper[s[r]] += 1  # must be 2 now
                while (
                    deduper[s[r]] > 1 and l < r
                ):  # "l < r" vs. "l <= r" does not matter - either will stop
                    deduper[s[l]] -= 1
                    l += 1

        return longest
```
