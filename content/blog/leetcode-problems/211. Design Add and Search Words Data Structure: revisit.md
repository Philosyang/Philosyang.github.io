class WordDictionary:

    def __init__(self):
        self.root = {}

    def addWord(self, word: str) -> None:
        node = self.root
        for ch in word:
            if ch not in node:
                node[ch] = {}
            node = node[ch]
        node["ended"] = True

    def search(self, word: str) -> bool:
        n = len(word)

        def dfs(node: dict, index: int) -> bool:
            if type(node) == bool:
                return False
            if index >= n:
                return node.get("ended", False)
            ch = word[index]
            if ch != ".":  # typical letter
                if node.get(ch, False):
                    return dfs(node[ch], index + 1)
                else:
                    return False
            else:  # wildcard
                for next_node in node.values():
                    if dfs(next_node, index + 1):
                        return True
                return False

        return dfs(self.root, 0)

class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False


class WordDictionary:

    def __init__(self):
        self.root = TrieNode()

    def addWord(self, word: str) -> None:
        node = self.root
        for ch in word:
            if ch not in node.children:
                node.children[ch] = TrieNode()
            node = node.children[ch]
        node.is_end = True

    def search(self, word: str) -> bool:
        n = len(word)

        def dfs(node, index: int) -> bool:
            if index >= n:
                return node.is_end

            ch = word[index]
            if ch == ".":
                for child in node.children.values():
                    if dfs(child, index + 1):
                        return True
                return False
            else:
                if ch in node.children:
                    return dfs(node.children[ch], index + 1)
                else:
                    return False

        return dfs(self.root, 0)

# obj = WordDictionary()
# obj.addWord(word)
# param_2 = obj.search(word)
