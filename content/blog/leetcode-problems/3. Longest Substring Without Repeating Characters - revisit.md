```python
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        # intuition: two pointer / sliding window
        result = 0
        l = 0
        r = -1
        n = len(s)
        last_seen = {}

        while r < n - 1:
            r += 1
            ch = s[r]
            if ch in last_seen and last_seen[ch] >= l:
                l = last_seen[ch] + 1

            last_seen[ch] = r
            result = max(result, r - l + 1)

        return result
```
