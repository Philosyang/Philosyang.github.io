```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(
        self, list1: Optional[ListNode], list2: Optional[ListNode]
    ) -> Optional[ListNode]:
        if not list1 and not list2:
            return None
        elif not list1:
            return list2
        elif not list2:
            return list1

        head = None
        if list1.val <= list2.val:
            head = list1
            list1 = list1.next
        else:
            head = list2
            list2 = list2.next

        node = head

        while True:
            nxt = None
            if list1 and list2:
                if list1.val <= list2.val:
                    nxt = list1
                    list1 = list1.next
                else:
                    nxt = list2
                    list2 = list2.next
            elif list1:
                nxt = list1
                list1 = list1.next
            elif list2:
                nxt = list2
                list2 = list2.next
            else:
                break

            node.next = nxt
            node = nxt

        return head
```
