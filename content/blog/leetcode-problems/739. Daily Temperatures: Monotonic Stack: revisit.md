```python
class Solution:
    def dailyTemperatures(self, temperatures: List[int]) -> List[int]:
        # monotonic stack: why?
        # we only need to know the next day that
        # has a higher temp than today

        n = len(temperatures)
        idx = [n - 1]
        result_reversed = [0]

        for i in range(n - 2, -1, -1):
            while idx and temperatures[i] >= temperatures[idx[-1]]:
                idx.pop()
            if idx:
                result_reversed.append(idx[-1] - i)
            else:
                result_reversed.append(0)
            idx.append(i)

        return result_reversed[::-1]
```
