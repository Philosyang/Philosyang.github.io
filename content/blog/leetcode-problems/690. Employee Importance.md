```python
"""
# Definition for Employee.
class Employee:
    def __init__(self, id: int, importance: int, subordinates: List[int]):
        self.id = id
        self.importance = importance
        self.subordinates = subordinates
"""


class Solution:
    def getImportance(self, employees: List["Employee"], id: int) -> int:

        employee_dict = {}

        for e in employees:
            employee_dict[e.id] = [e.importance, e.subordinates]

        dq = deque([id])
        result = 0

        while dq:
            importance, subordinates = employee_dict[dq.popleft()]
            result += importance

            for s in subordinates:
                dq.append(s)

        return result
```
