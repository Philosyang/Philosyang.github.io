```python
class Solution:
    def maxWeight(self, pizzas: List[int]) -> int:
        # intuition
        # greedy?
        # sort and only look at the larger half
        # day0: largest0
        # day1: largest2
        # day2: largest3
        # ...

        # close but:
        # for 3 (odd) days, we actually need the [-1], [-2], and [-4]
        # rather than the [-1], [-3], [-4]

        n = len(pizzas)
        days = n // 4
        odd_days, even_days = (days + 1) // 2, days // 2
        pizzas.sort()
        result = 0
        idx = 0

        for _ in range(odd_days):
            idx -= 1
            result += pizzas[idx]
        for _ in range(even_days):
            idx -= 2
            result += pizzas[idx]

        return result
```
