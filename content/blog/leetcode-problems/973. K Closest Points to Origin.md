```python
    def kClosest(self, points: List[List[int]], k: int) -> List[List[int]]:
        # heap will work
        heap = []
        for i, coord in enumerate(points):
            x, y = coord
            heapq.heappush(heap, ((x**2 + y**2) ** 0.5, i))  # (dist, idx in points)

        result = []
        for i in range(k):
            dist, i_points = heapq.heappop(heap)
            result.append(points[i_points])

        return result
```

we can also do a max-heap with capacity of k (we need max heap so that we can pop the furthest one without going through all the points).

```python
class Solution:
    def kClosest(self, points: List[List[int]], k: int) -> List[List[int]]:
        # heap will work
        heap = []
        for x, y in points:
            heapq.heappush(heap, (-(x * x + y * y), x, y))  # (dist**2, x, y)
            if len(heap) > k:
                heapq.heappop(heap)

        return [[x, y] for _, x, y in heap]
```
