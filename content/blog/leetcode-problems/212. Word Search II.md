```python
class Trie:

    def __init__(self):
        self.root = {}

    def insert(self, word: str) -> None:
        node = self.root
        for ch in word:
            if ch not in node:
                node[ch] = {}
            node = node[ch]
        node["word"] = word


class Solution:
    def findWords(self, board: List[List[str]], words: List[str]) -> List[str]:
        trie = Trie()
        for word in words:
            trie.insert(word)

        dirs = [(0, 1), (0, -1), (1, 0), (-1, 0)]
        m, n = len(board), len(board[0])
        result = set()

        def dfs(x: int, y: int, node: dict) -> None:
            ch = board[x][y]
            if ch not in node:
                return

            node = node[ch]
            if "word" in node:
                result.add(node["word"])

            board[x][y] = "#"  # visited
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if 0 <= nx < m and 0 <= ny < n:
                    dfs(nx, ny, node)
            board[x][y] = ch  # un-visit

            return

        for i in range(m):
            for j in range(n):
                dfs(i, j, trie.root)

        return list(result)
```
