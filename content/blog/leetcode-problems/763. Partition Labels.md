class Solution:
    def partitionLabels(self, s: str) -> List[int]:
        # idea:
        # for the 26 letters, find out each one's starting and ending: O(n)
        # one last traversal from left to right: cut when we have no in-betweens

        letters = {}  # [starting idx, ending idx]
        for idx, ch in enumerate(s):
            if ch not in letters:
                letters[ch] = [idx, idx]
            else:
                letters[ch][1] = idx
                
        # print(letters)

        result = []
        segment = 0
        unfinished_letter_count = 0

        for idx, ch in enumerate(s):
            # add current ch, then check unfinished + store to result
            segment += 1
            if idx == letters[ch][0]:  # if we see ch the first time
                unfinished_letter_count += 1
            if idx == letters[ch][1]:  # if idx is the last occurance of ch
                unfinished_letter_count -= 1

            if not unfinished_letter_count:  # if all are finished
                result.append(segment)
                segment = 0
            
            # print(idx, ch, result, segment, unfinished_letter_count)

        if segment:
            result.append(segment)

        return result
