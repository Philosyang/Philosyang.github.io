This is such a great question.

```python
# no matter what, rotating is the best choice (greedy)
# therefore, try to rotate as much as possible
# deque: [(A, 3, 0), (B, 3, 0)]; popleft, append
# A means task, 3 means 3 As left, 0 means the next possible non-wait task time
# O(n)
# the above idea is wrong: ["B","C","D","A","A","A","A","G"]
# ideally we should ABACADAG
# that means heapq, pop 2 largest (if we still have 2+)
# [(-3, A, 0), (-2, B, 0)]
# task name is actually optional
# above is still wrong, we actually need to pop n+1 largest
# and we actually don't need to track earliest_convenience
# each iteration of pop -> cpu_time += n+1

# follow-up: there is a O(n) math way


class Solution:
    def leastInterval(self, tasks: List[str], n: int) -> int:
        task_cnt = Counter(tasks)
        task_list = []
        for task, count in task_cnt.items():
            task_list.append(-count)
        heapq.heapify(task_list)
        cpu_time = 0
        last_round_cpu_time = 0

        while task_list:
            last_round_cpu_time = len(task_list)
            basket = []
            for i in range(min(last_round_cpu_time, n + 1)):
                basket.append(heapq.heappop(task_list))
            for j in range(len(basket)):
                task = basket[j]
                if task < -1:
                    heapq.heappush(task_list, task + 1)
            cpu_time += n + 1

        return cpu_time - n - 1 + last_round_cpu_time
```
