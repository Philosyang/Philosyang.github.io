```python
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        if not s:
            return 0

        n = len(s)
        l = 0
        r = -1
        pos = {}
        max_len = 1

        while r < n - 1:
            r += 1
            if pos.get(s[r], -1) >= l:
                l = pos[s[r]] + 1
            pos[s[r]] = r
            max_len = max(max_len, r - l + 1)

        return max_len
```
