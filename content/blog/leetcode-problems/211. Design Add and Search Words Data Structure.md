class WordDictionary:

    def __init__(self):
        self.root = {}

    def addWord(self, word: str) -> None:
        node = self.root
        for ch in word:
            if ch not in node:
                node[ch] = {}
            node = node[ch]
        node["ended"] = True

    def search(self, word: str) -> bool:
        def dfs(node, word) -> bool:
            if not node or type(node) == bool:
                return False
            if not word:
                return node.get("ended", False)

            ch = word[0]
            if ch == ".":
                for v in node.values():
                    if dfs(v, word[1:]):
                        return True
                return False
            else:  # if ch != '.'
                if ch not in node:
                    return False
                return dfs(node[ch], word[1:])

        return dfs(self.root, word)
