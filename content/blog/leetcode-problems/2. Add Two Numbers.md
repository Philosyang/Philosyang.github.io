# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(
        self, l1: Optional[ListNode], l2: Optional[ListNode]
    ) -> Optional[ListNode]:
        result = ListNode()
        node = result
        carry = 0
        prev = None

        while l1 and l2:
            add = l1.val + l2.val + carry
            if add >= 10:
                carry = 1
                add -= 10
            else:
                carry = 0
            node.val = add
            node.next = ListNode()
            prev = node
            node = node.next

            l1 = l1.next
            l2 = l2.next

        while l1:
            add = l1.val + carry
            if add >= 10:
                carry = 1
                add -= 10
            else:
                carry = 0
            node.val = add
            node.next = ListNode()
            prev = node
            node = node.next

            l1 = l1.next

        while l2:
            add = l2.val + carry
            if add >= 10:
                carry = 1
                add -= 10
            else:
                carry = 0
            node.val = add
            node.next = ListNode()
            prev = node
            node = node.next

            l2 = l2.next

        if carry:
            node.val = 1
            node.next = None
        else:
            prev.next = None

        return result
