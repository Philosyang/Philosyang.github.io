class Solution:
    def checkOverlap(
        self,
        radius: int,
        xCenter: int,
        yCenter: int,
        x1: int,
        y1: int,
        x2: int,
        y2: int,
    ) -> bool:
        # intuition:
        # start from circle center
        # find dist to 4 edges (incl. extensions) of the rectangle
        # true if (point not on extension) and (dist <= radius)
        # also true if circle center within rectangle
        # also true if 4 rectangle corners to circle center dist <= radius

        square_left = x1
        square_right = x2
        square_bottom = y1
        square_top = y2

        circle_x = xCenter
        circle_y = yCenter

        def euclidean_distance(ex0, ey0, ex1, ey1) -> float:
            return ((ex0 - ex1) ** 2 + (ey0 - ey1) ** 2) ** 0.5

        # left
        if abs(circle_x - square_left) <= radius and y1 <= circle_y <= y2:
            return True
        # right
        if abs(circle_x - square_right) <= radius and y1 <= circle_y <= y2:
            return True
        # bottom
        if abs(circle_y - square_bottom) <= radius and x1 <= circle_x <= x2:
            return True
        # top
        if abs(circle_y - square_top) <= radius and x1 <= circle_x <= x2:
            return True
        # within
        if x1 <= circle_x <= x2 and y1 <= circle_y <= y2:
            return True

        # bottom-left corner
        if euclidean_distance(circle_x, circle_y, x1, y1) <= radius:
            return True
        # bottom-right corner
        if euclidean_distance(circle_x, circle_y, x2, y1) <= radius:
            return True
        # top-left corner
        if euclidean_distance(circle_x, circle_y, x1, y2) <= radius:
            return True
        # top-right corner
        if euclidean_distance(circle_x, circle_y, x2, y2) <= radius:
            return True

        return False
