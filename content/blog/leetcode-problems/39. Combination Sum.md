remember to append the copy[:], as well as combinations vs. permutations (start index).

```python
class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        result = []
        n = len(candidates)

        def bt(start: int, r_cand: List[int], r_sum: int):
            if r_sum == target:
                result.append(r_cand[:])
            elif r_sum < target:
                for i in range(start, n):
                    cand_i = candidates[i]
                    r_cand.append(cand_i)
                    bt(i, r_cand, r_sum + cand_i)
                    r_cand.pop()

        bt(0, [], 0)

        return result
```
