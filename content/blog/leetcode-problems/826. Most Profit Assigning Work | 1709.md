class Solution:
    def maxProfitAssignment(
        self, difficulty: List[int], profit: List[int], worker: List[int]
    ) -> int:
        # intuition:
        # sort difficulty with key = profit
        # go through, if a tougher job has a lower profit del it
        # bi search for each worker

        diff_prof = []
        for i in range(len(difficulty)):
            diff_prof.append((profit[i], difficulty[i]))
        diff_prof.sort()

        meaningful_work = deque([(-1, -1)])
        for p, d in diff_prof:
            while d <= meaningful_work[-1][1]:
                meaningful_work.pop()
            meaningful_work.append((p, d))
        meaningful_work.popleft()

        ndiff = []
        nprof = []
        for p, d in meaningful_work:
            ndiff.append(d)
            nprof.append(p)

        result = 0
        for w in worker:
            i = bisect_right(ndiff, w)
            if i != 0:
                result += nprof[i - 1]

        return result
