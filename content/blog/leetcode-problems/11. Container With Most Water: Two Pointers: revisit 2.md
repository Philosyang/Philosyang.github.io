```python
class Solution:
    def maxArea(self, height: List[int]) -> int:
        # idea: two pointer
        if not height or len(height) < 2:
            return 0

        n = len(height)
        li, ri = 0, n - 1
        mw = min(height[li], height[ri]) * (ri - li)

        # def update(li, ri):
        #     nonlocal mw
        #     w = min(height[li], height[ri]) * (ri - li)
        #     if w > mw:
        #         mw = w
        #
        # while li < ri:
        #     if height[li] >= height[ri]:
        #         while height[li] >= height[ri] and li < ri:
        #             ri -= 1
        #             if height[ri] > height[ri + 1]:
        #                 update(li, ri)
        #     else:
        #         while height[li] < height[ri] and li < ri:
        #             li += 1
        #             if height[li] > height[li - 1]:
        #                 update(li, ri)

        while li < ri:
            mw = max(mw, min(height[li], height[ri]) * (ri - li))
            if height[li] < height[ri]:
                li += 1
            else:
                ri -= 1

        return mw
```
