```python
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        # idea: backtrack, carry a list of remaining unused nums
        result = []

        def bt(ans: List[int], remains: List[int]):
            nonlocal result

            if not remains:
                result.append(ans)
                return

            for i, num in enumerate(remains):
                ans.append(num)
                bt(ans.copy(), remains[:i] + remains[i + 1 :])
                ans.pop()

        bt([], nums)

        return result
```
