```python
class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        count = Counter(nums)
        freq = sorted([(cnt, num) for num, cnt in count.items()])
        result = []
        for i in range(1, k + 1):
            result.append(freq[-i][1])
        return result
```
