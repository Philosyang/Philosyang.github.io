```python
class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        # startswith
        n = len(s)

        @lru_cache(None)
        def wb(i: int) -> bool:
            if i == n:
                return True
            # if i > n:
            #     return False

            for word in wordDict:
                if s[i:].startswith(word):
                    if wb(i + len(word)):
                        return True

            return False

        return wb(0)
```
