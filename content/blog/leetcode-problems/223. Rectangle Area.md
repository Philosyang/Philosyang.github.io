```python
class Solution:
    def computeArea(self, ax1: int, ay1: int, ax2: int, ay2: int, bx1: int, by1: int, bx2: int, by2: int) -> int:
        maxleft = max(ax1, bx1)
        minright = min(ax2, bx2)
        maxbottom = max(ay1, by1)
        mintop = min(ay2, by2)

        overlap = 0

        if maxleft >= minright or maxbottom >= mintop:  # no overlap
            overlap = 0
        else:
            overlap = (minright - maxleft) * (mintop - maxbottom)

        return (ax2 - ax1) * (ay2 - ay1) + (bx2 - bx1) * (by2 - by1) - overlap
```
