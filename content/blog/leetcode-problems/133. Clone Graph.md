```python
"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

from typing import Optional


class Solution:
    def cloneGraph(self, node: Optional["Node"]) -> Optional["Node"]:
        # idea: given values are unique, hashmap `values: cloned_node`

        if not node:
            return None

        dq = deque([node])
        clones = {node.val: Node(node.val, [])}

        while dq:
            original = dq.popleft()
            clone = clones[original.val]
            clone_nei = []

            for nei in original.neighbors:
                if nei.val not in clones:
                    clones[nei.val] = Node(nei.val, [])
                    dq.append(nei)
                clone_nei.append(clones[nei.val])
            clone.neighbors = clone_nei

        return clones[node.val]
```
