Canonical idea: fast/slow pointers.

```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        # canonical idea:
        # fast/slow pointers - fast will be n node faster

        fast = slow = head
        for i in range(n):
            fast = fast.next
        
        # special case
        if not fast:
            return head.next

        while fast.next:
            slow = slow.next
            fast = fast.next
        
        prev = slow
        prev.next = slow.next.next

        return head
```
