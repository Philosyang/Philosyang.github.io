```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head:
            return None

        buffer = deque([])
        node = head

        while node:
            buffer.appendleft(node)
            node = node.next

        # print(buffer)

        head = buffer[0]
        node = head

        for i, node in enumerate(buffer):
            if i == len(buffer) - 1:
                node.next = None
            else:
                node.next = buffer[i + 1]

        return head
```

This version is passing with tc=O(2n), sc=O(n).

The clean, standard iterative approach is O(n), O(1).

```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        prev = None
        node = head

        # 1 -> 2 -> 3
        while node:
            nxt = node.next     # 2
            node.next = prev    # 0
            prev = node         # 1
            node = nxt          # 2
        
        return prev
```
