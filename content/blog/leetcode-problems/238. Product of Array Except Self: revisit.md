class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        # idea: two arrays
        # pre_product: product of everything before me
        # post_product: product of everything after me

        n = len(nums)

        pre_prod = [1]
        post_prod_reversed = [1]

        for i in range(n):
            pre_prod.append(pre_prod[-1] * nums[i])
        for j in range(n - 1, -1, -1):
            post_prod_reversed.append(post_prod_reversed[-1] * nums[j])

        post_prod = post_prod_reversed[::-1]
        # print(pre_prod, post_prod)

        result = []

        for k in range(n):
            result.append(pre_prod[k] * post_prod[k + 1])

        return result
